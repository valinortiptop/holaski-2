\n?/g, "").trim();
        results = JSON.parse(cleaned);
      } catch {
        results = [];
      }

      return new Response(JSON.stringify({ results }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "ai-chat") {
      const { messages } = params;
      const systemMessage = {
        role: "system",
        content: `Eres el asistente de viajes de HolaSki, un experto en viajes de esquí. Ayudas a los usuarios a planificar su viaje perfecto de esquí. Eres amigable, conocedor y entusiasta. Responde siempre en español. Cuando el usuario te dé suficiente información (destino, fechas, presupuesto, nivel), genera una recomendación detallada. Si el usuario pide un paquete, incluye un JSON al final de tu respuesta con el formato:
###PACKAGE_JSON###
{
  "resort": "nombre",
  "country": "país",
  "dates": "fechas",
  "guests": "número",
  "hotel": "nombre hotel sugerido",
  "price_per_person": "precio USD",
  "total_price": "precio total USD",
  "includes": ["vuelo", "hotel", "forfait", "traslados"],
  "skill_level": "nivel"
}
###END_PACKAGE_JSON###
Solo incluye el JSON cuando tengas suficiente info para armar un paquete.`,
      };

      const res = await fetch(PROXY_URL, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-proxy-token": PROXY_TOKEN,
        },
        body: JSON.stringify({
          provider: "openai",
          endpoint: "/v1/chat/completions",
          payload: {
            model: "gpt-4o-mini",
            messages: [systemMessage, ...messages],
            temperature: 0.7,
          },
        }),
      });

      const data = await res.json();
      const content = data?.choices?.[0]?.message?.content || "Lo siento, no pude procesar tu solicitud.";

      return new Response(JSON.stringify({ message: content }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "send-contact-email") {
      const { name, email, phone, message } = params;

      const res = await fetch(PROXY_URL, {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "x-proxy-token": PROXY_TOKEN,
        },
        body: JSON.stringify({
          provider: "resend",
          endpoint: "/emails",
          payload: {
            from: "HolaSki <onboarding@resend.dev>",
            to: ["delivered@resend.dev"],
            subject: `Nuevo contacto de ${name}`,
            html: `<h2>Nuevo mensaje de contacto</h2><p><strong>Nombre:</strong> ${name}</p><p><strong>Email:</strong> ${email}</p><p><strong>Teléfono:</strong> ${phone || "No proporcionado"}</p><p><strong>Mensaje:</strong></p><p>${message}</p><hr/><p style="font-size:12px;color:#666;">Para dejar de recibir estos emails, <a href="https://holaski.com/unsubscribe?email=${encodeURIComponent(email)}">haz clic aquí</a>.</p>`,
          },
        }),
      });

      const data = await res.json();
      return new Response(JSON.stringify({ success: true, data }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "subscribe-newsletter") {
      const { email } = params;

      try {
        await fetch(PROXY_URL, {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "x-proxy-token": PROXY_TOKEN,
          },
          body: JSON.stringify({
            provider: "resend",
            endpoint: "/emails",
            payload: {
              from: "HolaSki <onboarding@resend.dev>",
              to: [email],
              subject: "¡Bienvenido a HolaSki! 🎿",
              html: `<div style="font-family:sans-serif;max-width:600px;margin:0 auto;"><h1 style="color:#2563eb;">¡Bienvenido a HolaSki!</h1><p>Gracias por suscribirte a nuestro newsletter. Recibirás las mejores ofertas de esquí directamente en tu bandeja de entrada.</p><p>¡Nos vemos en las pistas! ⛷️</p><hr/><p style="font-size:12px;color:#666;">Si no deseas recibir más correos, <a href="https://holaski.com/unsubscribe?email=${encodeURIComponent(email)}">cancela tu suscripción</a>.</p></div>`,
            },
          }),
        });
      } catch (e) {
        console.log("Email send failed but continuing:", e);
      }

      return new Response(JSON.stringify({ success: true }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "unsubscribe") {
      const { email } = params;
      return new Response(JSON.stringify({ success: true, email }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    return new Response(JSON.stringify({ error: "Unknown action: " + action }), {
      status: 400,
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  } catch (err) {
    console.error("api-handler error:", err);
    return new Response(JSON.stringify({ error: "Internal server error" }), {
      status: 500,
      headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  }
});

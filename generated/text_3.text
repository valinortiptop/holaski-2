\n?/g, "").trim();
        results = JSON.parse(cleaned);
      } catch { results = []; }
      return new Response(JSON.stringify({ results }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "ai-chat") {
      const { messages } = params;
      const res = await fetch(PROXY_URL, {
        method: "POST",
        headers: { "Content-Type": "application/json", "x-proxy-token": PROXY_TOKEN },
        body: JSON.stringify({
          provider: "openai",
          endpoint: "/v1/chat/completions",
          payload: {
            model: "gpt-4o-mini",
            messages: [
              { role: "system", content: "Eres el asistente AI de HolaSki, experto en viajes de esquí. Responde SIEMPRE en español. Sé amigable, entusiasta y útil. Ayuda a planificar viajes de esquí perfectos. Pregunta nivel, presupuesto, fechas y preferencias. Sugiere destinos concretos con precios estimados." },
              ...messages
            ],
            temperature: 0.7
          },
        }),
      });
      const data = await res.json();
      const content = data?.choices?.[0]?.message?.content || "Lo siento, hubo un error. Intenta de nuevo.";
      return new Response(JSON.stringify({ message: content }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "send-contact-email") {
      const { name, email, phone, message } = params;
      const res = await fetch(PROXY_URL, {
        method: "POST",
        headers: { "Content-Type": "application/json", "x-proxy-token": PROXY_TOKEN },
        body: JSON.stringify({
          provider: "resend",
          endpoint: "/emails",
          payload: {
            from: "HolaSki <onboarding@resend.dev>",
            to: ["delivered@resend.dev"],
            subject: `Nuevo contacto: ${name}`,
            html: `<h2>Nuevo contacto de HolaSki</h2><p><b>Nombre:</b> ${name}</p><p><b>Email:</b> ${email}</p><p><b>Teléfono:</b> ${phone || "N/A"}</p><p><b>Mensaje:</b> ${message}</p><hr/><p style="font-size:11px;color:#999;"><a href="https://holaski.com/unsubscribe?email=${encodeURIComponent(email)}">Cancelar suscripción</a></p>`
          },
        }),
      });
      const data = await res.json();
      return new Response(JSON.stringify({ success: true, data }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "subscribe-newsletter") {
      const { email } = params;
      try {
        await fetch(PROXY_URL, {
          method: "POST",
          headers: { "Content-Type": "application/json", "x-proxy-token": PROXY_TOKEN },
          body: JSON.stringify({
            provider: "resend",
            endpoint: "/emails",
            payload: {
              from: "HolaSki <onboarding@resend.dev>",
              to: [email],
              subject: "¡Bienvenido a HolaSki! 🎿",
              html: `<h1>¡Bienvenido a HolaSki!</h1><p>Gracias por suscribirte. Recibirás ofertas exclusivas.</p><hr/><p style="font-size:11px;color:#999;"><a href="https://holaski.com/unsubscribe?email=${encodeURIComponent(email)}">Cancelar suscripción</a></p>`
            },
          }),
        });
      } catch (e) { console.log("Newsletter email failed:", e); }
      return new Response(JSON.stringify({ success: true }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    if (action === "unsubscribe") {
      return new Response(JSON.stringify({ success: true }), {
        headers: { ...corsHeaders, "Content-Type": "application/json" },
      });
    }

    return new Response(JSON.stringify({ error: "Unknown action" }), {
      status: 400, headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  } catch (err) {
    console.error("api-handler error:", err);
    return new Response(JSON.stringify({ error: "Internal error" }), {
      status: 500, headers: { ...corsHeaders, "Content-Type": "application/json" },
    });
  }
});

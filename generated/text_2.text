")) s = s.slice(0, -3)
  return s.trim()
}

async function handleAiSearch(query: string, proxyUrl: string, proxyToken: string) {
  const systemPrompt = `Eres un experto asesor de viajes de esqui. Devuelve SOLO JSON valido.
Formato: {"results":[{"id":"slug","resort_name":"Nombre","country":"Pais","region":"Region","price_range_usd":"$150-$300/dia","difficulty":"beginner|intermediate|advanced","highlights":["Punto 1"],"rating":4.8,"image_url":"URL","match_score":95,"why_it_matches":"Explicacion"}]}
Usa estas imagenes:
1. https://images.unsplash.com/photo-1551524559-8af4e6624178?w=600
2. https://images.unsplash.com/photo-1565992441121-4367c2967103?w=600
3. https://images.unsplash.com/photo-1519681393784-d120267933ba?w=600`

  try {
    const res = await fetch(proxyUrl, {
      method: "POST",
      headers: { "Content-Type": "application/json", "x-proxy-token": proxyToken },
      body: JSON.stringify({
        provider: "openai",
        endpoint: "/v1/chat/completions",
        payload: {
          model: "gpt-5.3-chat-latest",
          messages: [{ role: "system", content: systemPrompt }, { role: "user", content: query }],
          temperature: 0.7
        }
      })
    })

    if (!res.ok) throw new Error("Proxy error")
    const data = await res.json()
    const content = stripFences(data.choices[0].message.content)
    return new Response(content, { headers: corsHeaders })
  } catch {
    return new Response(JSON.stringify({ results: getFallbackResorts() }), { headers: corsHeaders })
  }
}

async function handleSendContact(body: any, proxyUrl: string, proxyToken: string) {

      body: JSON.stringify({

        endpoint: "/emails",

          from: "HolaSki <onboarding@resend.dev>",

          subject: "Contacto: " + body.name,
          html: `<p>Nombre: ${body.name}</p><p>Email: ${body.email}</p><p>Mensaje: ${body.message}</p>`
        }
      })
    })
    const data = await res.json()

  } catch (err) {
    return new Response(JSON.stringify({ success: false, error: String(err) }), { headers: corsHeaders })
  }
}
